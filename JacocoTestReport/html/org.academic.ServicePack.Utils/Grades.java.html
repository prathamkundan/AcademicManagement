package org.academic.ServicePack.Utils;

public enum Grades {
    A(10, "A"),
    AM(9, "A-"),
    B(8, "B"),
    BM(7, "B-"),
    C(6, "C"),
    CM(5, "C-"),
    D(4, "D"),
    DM(3, "D-"),
    E(2, "E"),
    EM(1, "E-"),
    F(0, "F"),
    NA(0, "NA");

    public final String grade;
    public final int value;

    Grades(int value, String grade) {
        this.value = value;
        this.grade = grade;
    }

    public static Grades get(String grade) {
        for (Grades g : Grades.values()) {
            if (g.grade.equals(grade)) {
                return g;
            }
        }
        return null;
    }
    public static Grades get(int grade) {
        for (Grades g : Grades.values()) {
            if (g.value == grade) {
                return g;
            }
        }
        return null;
    }
}
