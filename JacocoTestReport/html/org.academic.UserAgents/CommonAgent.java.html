package org.academic.UserAgents;

import org.academic.ServicePack.CommonInterface;
import org.academic.ServicePack.DataObjects.CourseObject;
import org.academic.ServicePack.DataObjects.RecordObject;
import org.academic.ServicePack.DataObjects.SessionObject;
import org.academic.ServicePack.DataObjects.UserObject;
import org.academic.ServicePack.Utils.AppException;
import org.academic.ServicePack.Utils.CourseType;
import org.academic.ServicePack.Utils.Grades;
import org.academic.ServicePack.Utils.Roles;

import java.util.ArrayList;
import java.util.HashMap;

public class CommonAgent {
    CommonInterface commonService;
    public CommonAgent(CommonInterface commonService) {
        this.commonService = commonService;
    }

    public SessionObject getCurrentSemesterData() throws AppException {
        try {
            return commonService.getCurrentSemesterData();
        } catch (AppException ae) {
            throw new AppException("Could not get current semester data", ae);
        }
    }

    public ArrayList<SessionObject> getAllSemesterData() throws AppException {
        try {
            return commonService.getAllSemesterData();
        } catch (AppException ae) {
            throw new AppException("Could not get all semester data", ae);
        }
    }

    public UserObject getUserObject(String user_id) throws AppException {
        try {
            return commonService.getStudentObject(user_id);
        } catch (AppException ignored) {}
        try {
            return commonService.getFacultyObject(user_id);
        } catch (AppException ignored) {}
        try {
            return commonService.getAdminObject(user_id);
        } catch (AppException e) {
            throw new AppException("Could not find user: " + user_id);
        }
    }

    public HashMap<CourseObject, Integer> getCourseConstraints(String course_id, int year, int semester) throws AppException {
        try {
            return commonService.getCourseConstraints(course_id, year, semester);
        } catch (AppException ae) {
            throw new AppException("Could not get course constraints", ae);
        }
    }

    public ArrayList<CourseObject> getCoursePrerequisites(String course_id) throws AppException {
        try {
            return commonService.getCoursePrerequisites(course_id);
        } catch (AppException ae) {
            throw new AppException("Could not get course prerequisites", ae);
        }
    }

    public CourseObject getCourseObject(String course_id) throws AppException {
        try {
            return commonService.getCourseObject(course_id);
        } catch (AppException ae) {
            throw new AppException("Could not get course data", ae);
        }
    }

    public ArrayList<CourseObject> getAllCourses() throws AppException {
        try {
            return commonService.getAllCourses();
        } catch (AppException ae) {
            throw new AppException("Could not get all courses", ae);
        }
    }

    public ArrayList<CourseObject> getCoreCourses(int batch, String dept) throws AppException {
        try {
            return commonService.getCoreCourses(batch, dept);
        } catch (AppException ae) {
            throw new AppException("Could not get core courses", ae);
        }
    }

    public HashMap<CourseType, Integer> getUGCurriculum(int batch) throws AppException {
        try {
            return commonService.getUGCurriculum(batch);
        } catch (AppException ae) {
            throw new AppException("Could not get UG curriculum", ae);
        }
    }

    public void updateInfo(UserObject newUserData, String password) throws AppException {
        try {
            commonService.updateInfo(newUserData, password);
        } catch (AppException ae) {
            throw new AppException("Could not update user data", ae);
        }
    }

    public ArrayList<RecordObject> getCourseRecord(String course_id, int year, int semester) throws AppException {
        try {
            return commonService.getCourseRecord(course_id, year, semester);
        } catch (AppException ae) {
            throw new AppException("Could not get course record", ae);
        }
    }

    public ArrayList<RecordObject> getSemesterRecords(String student_id, int year, int semester) throws AppException {
        try {
            ArrayList<CourseObject> courses = commonService.getEnrolledCourses(student_id, year, semester);
            ArrayList<RecordObject> records = new ArrayList<>();
            for (CourseObject course : courses) records.add(commonService.getRecordObject(student_id, course.course_id, year, semester));
            return records;
        } catch (AppException e) {
            throw new AppException("Could not get semester records", e);
        }
    }

    public double getSGPA(String student_id, int year, int semester) throws AppException {
        try {
            double cum_credits = 0;
            double cum_grade = 0;
            ArrayList<CourseObject> enrolled_courses = commonService.getEnrolledCourses(student_id, year, semester);
            for (CourseObject course : enrolled_courses) {
                int offering_id = commonService.getOfferingId(course.course_id, year, semester);

                cum_credits += course.credits;
                cum_grade += commonService.getGrade(student_id, offering_id).value * course.credits;
            }
            return cum_grade/cum_credits;
        } catch (AppException e){
            throw new AppException("Could not get SGPA", e);
        }
    }

    public double getCGPA(String student_id) throws AppException {
        try {
            double cum_grade = 0.0;
            double cum_credits = 0.0;

            ArrayList<CourseObject> all_courses = getAllCourses();
            for (CourseObject course : all_courses) {
                Grades grade = commonService.getGrade(student_id, course.course_id);
                if (grade.value > 4) {
                    cum_credits += course.credits;
                    cum_grade += grade.value * course.credits;
                }
            }

            if (cum_credits == 0) return 0.0;
            return cum_grade / cum_credits;
        } catch (AppException e) {
            throw new AppException("Could not get CGPA", e);
        }
    }

    public UserObject loginUser(String user_id, Roles role, String password) throws AppException {
        try {
            if (commonService.verifyUser(user_id, role, password)){
                commonService.logAction(user_id, "LOGIN");
                return getUserObject(user_id);
            } else {
                throw new AppException("Invalid credentials");
            }
        } catch (AppException ae) {
            throw new AppException("Could not login user", ae);
        }
    }

    public void logoutUser(String user_id) throws AppException {
        try {
            commonService.logAction(user_id, "LOGOUT");
        } catch (AppException ae) {
            throw new AppException("Could not logout user", ae);
        }
    }
}
