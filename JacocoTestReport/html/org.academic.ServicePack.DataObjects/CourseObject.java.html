package org.academic.ServicePack.DataObjects;

public class CourseObject {
    public String course_id;
    public String course_name;
    public String ltp;
    public String dept;
    public int credits;

    public CourseObject(String course_id, String course_name, String dept, String ltp, int credits) {
        this.course_id = course_id;
        this.course_name = course_name;
        this.ltp = ltp;
        this.credits = credits;
        this.dept = dept;
    }

    public String format(Boolean full) {
        String output = "";
        output = output + (String.format("%"+8+"s",this.course_id));
        output = output + (String.format("%"+40+"s",this.course_name));
        if (full) {
            output = output + (String.format("%"+8+"s",this.ltp));
            output = output + (String.format("%"+10+"d",this.credits));
        }
        return output;
    }

    public static String header(Boolean full) {
        String output = "";
        output = output + (String.format("%"+8+"s","ID"));
        output = output + (String.format("%"+40+"s","Course Name"));
        if (full) {
            output = output + (String.format("%"+8+"s","LTP"));
            output = output + (String.format("%"+10+"s","Credits"));
        }
        return output;
    }
}
