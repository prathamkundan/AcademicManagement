package org.academic.ServicePack.DataObjects;

import org.academic.ServicePack.Utils.CourseType;
import org.academic.ServicePack.Utils.Grades;

public class RecordObject {
    public final FacultyObject faculty;
    public final StudentObject student;
    public final CourseObject course;
    public final Grades grade;
    public final CourseType credit_option;

    public RecordObject(CourseObject course, StudentObject student, FacultyObject faculty, Grades grade, CourseType credit_option) {
        this.faculty = faculty;
        this.student = student;
        this.course = course;
        this.grade = grade;
        this.credit_option = credit_option;
    }

    public String format(Boolean showStudent) {
        String output = "";
        if (showStudent) output += (String.format("%"+10+"s", student.user_id));
        output += course.format(false);
        output += (String.format("%"+10+"s",faculty.user_id));
        output += (String.format("%"+5+"s",grade == null ? "-" : grade.grade));
        output += (String.format("%"+16+"s",credit_option));
        return output;
    }

    public static String header(Boolean showStudent) {
        String header = "";
        if (showStudent) header += (String.format("%"+10+"s", "Student"));
        header += CourseObject.header(false);
        header +=(String.format("%"+10+"s","Faculty"));
        header +=(String.format("%"+8+"s","Grade"));
        header +=(String.format("%"+16+"s","Credit Option"));
        return header;
    }
}
