package org.academic.ServicePack.DataObjects;

import org.academic.ServicePack.Utils.Roles;

public abstract class UserObject {
    public String user_id;
    public String user_name;
    public String email;
    public String phone;
    public Roles role;

    public UserObject(String user_id, String user_name, String email, String phone, Roles role) {
        this.user_id = user_id;
        this.user_name = user_name;
        this.email = email;
        this.phone = phone;
        this.role = role;
    }

    public abstract String getDepartment();

    public abstract int getBatch();

    public String format(){
        String output = "";
        output += this.role + ":" + "\n";
        output += "ID\t\t: " + this.user_id + "\n";
        output += "Name\t: " + this.user_name + "\n";
        output += "Email\t: " + this.email + "\n";
        output += "Phone\t: " + (this.phone.isBlank() ? "Not Provided" : this.phone) + "\n";
        if (this.getDepartment() != null) output += "Dept\t: " + this.getDepartment()+"\n";
        if (this.getBatch() != 0) output += "Batch\t: " + this.getBatch()+"\n";
        return output;
    }
}
