package org.academic.Agents;

import org.academic.Agents.AdminAgent;
import org.academic.Agents.FacultyAgent;
import org.academic.Agents.StudentAgent;
import org.academic.ServicePack.Utils.AppException;
import org.academic.Services.AdminService;
import org.academic.Services.FacultyService;
import org.academic.Services.StudentService;

public class AgentFactory {
    private final String url;
    private final String username;
    private final String password;
    public AgentFactory(String url, String username, String password) {
        this.url = url;
        this.username = username;
        this.password = password;
    }
    public StudentAgent getStudentAgent() throws AppException {
        return new StudentAgent(new StudentService(url, username, password));
    }

    public FacultyAgent getFacultyAgent() throws AppException {
        return new FacultyAgent(new FacultyService(url, username, password));
    }

    public AdminAgent getAdminAgent() throws AppException {
        return new AdminAgent(new AdminService(url, username, password));
    }
}
