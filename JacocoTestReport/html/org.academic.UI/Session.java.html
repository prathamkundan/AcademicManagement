package org.academic.UI;

import org.academic.UserAgents.AgentFactory;
import org.academic.Constants;
import org.academic.ServicePack.Utils.AppException;

public class Session {
    private final CLI CLI;
    AgentFactory agentFactory = new AgentFactory(Constants.DB_URL, Constants.DB_USERNAME, Constants.DB_PASSWORD);
    public Session(CLI cli) {
        this.CLI = cli;
    }
    public void start() throws AppException {
        CLI.dBreak();
        CLI.dString("-------- Welcome to Academic Management System --------");
        CLI.dBreak();

        while (true) {
            CLI.dString("Please select your role:");
            CLI.dString("1. Admin");
            CLI.dString("2. Faculty");
            CLI.dString("3. Student");
            CLI.dString("0. Exit");
            int choice = CLI.scanInt("Choice:");
            UserUI userCLI;
            switch (choice) {
                case 0 -> { return; }
                case 1 -> userCLI = new AdminUI(CLI, agentFactory);
                case 2 -> userCLI = new FacultyUI(CLI, agentFactory);
                case 3 -> userCLI = new StudentUI(CLI, agentFactory);
                default -> {
                    CLI.dString("Invalid choice");
                    continue;
                }
            }
            userCLI.displayActionMenu();
        }
    }

}
