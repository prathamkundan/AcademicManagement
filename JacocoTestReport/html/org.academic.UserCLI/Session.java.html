package org.academic.UserCLI;

import org.academic.Agents.AgentFactory;
import org.academic.CLI;
import org.academic.Constants;
import org.academic.ServicePack.Utils.AppException;

public class Session {
    private final CLI CLI;
    public Session(CLI cli) {
        this.CLI = cli;
    }
    private BaseUser userCLI;
    AgentFactory agentFactory = new AgentFactory(Constants.TEST_DB_URL, Constants.DB_USERNAME, Constants.DB_PASSWORD);
    public void start() throws AppException {
        CLI.dBreak();
        CLI.dString("-------- Welcome to Academic Management System --------");
        CLI.dBreak();

        while (true) {
            CLI.dString("Please select your role:");
            CLI.dString("1. Admin");
            CLI.dString("2. Faculty");
            CLI.dString("3. Student");
            CLI.dString("0. Exit");
            int choice = CLI.scanInt("Choice:");
            switch (choice) {
                case 0 -> { return; }
                case 1 -> userCLI = new Admin(CLI, agentFactory);
                case 2 -> userCLI = new Faculty(CLI, agentFactory);
                case 3 -> userCLI = new Student(CLI, agentFactory);
                default -> CLI.dString("Invalid choice");
            }
            userCLI.displayActionMenu();
        }
    }

}
